## 📋 Description

Retrieve all available quiz categories with metadata. Public endpoint accessible to all authenticated users. Categories are used to organize and filter quizzes.

## 🗏️ Business Logic (9 Steps)

1. **Validate Authentication**: Verify JWT token is valid, extract user_id, return 401 if invalid
2. **Build Base Query**: `SELECT category_id, name, description, icon_url, color_code, created_at FROM quiz_categories WHERE is_active=true AND is_deleted=false`
3. **Optionally Include Stats**: If `include_stats=true`, LEFT JOIN to `quizzes` table and COUNT quiz_id per category
4. **Apply Sorting**: If `sort_by="name"` order by name ASC; if `sort_by="quiz_count"` and stats included, order by count DESC
5. **Execute Query**: Fetch from database, handle null quiz_count as 0
6. **Format Response**: Map database results to response schema with all required fields
7. **Set Cache Headers**: Add `Cache-Control: public, max-age=3600` for 1-hour caching
8. **Validate Response Size**: Ensure payload < 50KB, return error if exceeds
9. **Log Access**: Log endpoint access with user_id and timestamp for audit trail

### Service Layer

```python
# app/services/quiz_category_service.py
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, func, and_
from app.models import QuizCategory, Quiz
from typing import List, Optional
import logging

logger = logging.getLogger(__name__)

class QuizCategoryService:
    @staticmethod
    async def get_all_categories(
        db: AsyncSession,
        include_stats: bool = False,
        sort_by: str = "name"
    ) -> List[dict]:
        """Retrieve all active quiz categories with optional stats"""
        try:
            query = select(QuizCategory).where(
                and_(
                    QuizCategory.is_active == True,
                    QuizCategory.is_deleted == False
                )
            )
            
            if sort_by == "name":
                query = query.order_by(QuizCategory.name)
            elif sort_by == "quiz_count" and include_stats:
                query = query.outerjoin(Quiz).group_by(QuizCategory.id)
                query = query.order_by(func.count(Quiz.id).desc())
            
            result = await db.execute(query)
            categories = result.scalars().all()
            
            response_data = []
            for cat in categories:
                cat_dict = {
                    "category_id": cat.id,
                    "name": cat.name,
                    "description": cat.description,
                    "icon_url": cat.icon_url,
                    "color_code": cat.color_code,
                    "created_at": cat.created_at
                }
                if include_stats:
                    quiz_count = await db.execute(
                        select(func.count(Quiz.id)).where(Quiz.category_id == cat.id)
                    )
                    cat_dict["quiz_count"] = quiz_count.scalar() or 0
                response_data.append(cat_dict)
            
            logger.info(f"Retrieved {len(response_data)} categories")
            return response_data
        except Exception as e:
            logger.error(f"Category retrieval error: {str(e)}", exc_info=True)
            raise
```

### API Handler

```python
# app/api/v1/quiz_categories.py
from fastapi import APIRouter, Depends, HTTPException, status, Query
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import get_db
from app.dependencies import get_current_user
from app.models import User
from app.services.quiz_category_service import QuizCategoryService
from app.schemas.quiz_category import CategoryListResponse
import logging

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/api/v1/quiz-categories", tags=["Quiz Categories"])

@router.get("", response_model=CategoryListResponse, status_code=status.HTTP_200_OK)
async def get_all_categories(
    include_stats: bool = Query(False),
    sort_by: str = Query("name", regex="^(name|quiz_count)$"),
    current_user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db)
) -> CategoryListResponse:
    """Get all active quiz categories"""
    try:
        categories = await QuizCategoryService.get_all_categories(
            db, include_stats=include_stats, sort_by=sort_by
        )
        return CategoryListResponse(
            success=True,
            data={"categories": categories}
        )
    except Exception as e:
        logger.error(f"Error: {str(e)}")
        raise HTTPException(status_code=500, detail="Failed to retrieve categories")
```
