## 📋 Description

Create a new quiz category with metadata. Only admin users can create categories. Validates unique category names and color codes.

---

## 🎯 API Specification

### Endpoint
```
POST /api/v1/quiz-categories
```

```json

  "name": "Geometry",
  "description": "Shapes, areas, volumes, and geometric theorems",
  "icon_url": "https://cdn.example.com/geometry.png",
```

```python
{
  "name": "required, string, 1-100 chars, unique",
  "description": "optional, string, 0-500 chars",
  "icon_url": "optional, valid HTTP/HTTPS URL",
  "color_code": "optional, regex ^#[0-9A-Fa-f]{6}$, default #000000"
```

## 🗏️ Business Logic (10 Steps)

1. **Validate Authentication**: Verify JWT token valid, extract user_id
2. **Check Admin Role**: Query users table, verify role='admin', return 403 if not admin
3. **Validate Input**: Use Pydantic schema to validate all required/optional fields
4. **Check Name Uniqueness**: Query `quiz_categories WHERE name ILIKE :name`, return 409 if exists
5. **Validate URL**: If icon_url provided, verify is valid HTTP/HTTPS URL using validators
6. **Validate Color Code**: If provided, match regex `^#[0-9A-Fa-f]{6}$`, return 400 if invalid
7. **Set Defaults**: Set `is_active=true`, `is_deleted=false`, `created_by_id=user_id`
8. **Insert Record**: `INSERT INTO quiz_categories (...) VALUES (...)`
9. **Cache Invalidation**: Delete Redis key `categories:all` to invalidate list cache
10. **Audit Log**: Log creation with user_id, category_id, timestamp to audit table

```python
# app/schemas/quiz_category.py
from pydantic import BaseModel, Field, HttpUrl, validator
from typing import Optional

class CategoryCreate(BaseModel):
    name: str = Field(..., min_length=1, max_length=100)
    description: Optional[str] = Field(None, max_length=500)
    icon_url: Optional[HttpUrl] = None
    color_code: Optional[str] = Field(
        default="#000000",
        regex=r"^#[0-9A-Fa-f]{6}$"
    )
    
    @validator('name')
    def name_not_empty(cls, v):
        if not v.strip():
            raise ValueError('Name cannot be empty or whitespace')
        return v.strip()

class CategoryCreateResponse(BaseModel):
    success: bool = True
    data: dict
```

### Service Layer

```python
# app/services/quiz_category_service.py
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.models import QuizCategory
from app.schemas.quiz_category import CategoryCreate
from fastapi import HTTPException, status
import logging
import re

logger = logging.getLogger(__name__)

class QuizCategoryService:
    @staticmethod
    async def create_category(
        db: AsyncSession,
        category_data: CategoryCreate,
        user_id: int,
        redis_client=None
    ) -> dict:
        """Create new quiz category"""
        try:
            # Check uniqueness
            existing = await db.execute(
                select(QuizCategory).where(
                    QuizCategory.name.ilike(category_data.name)
                )
            )
            if existing.scalar():
                raise HTTPException(
                    status_code=status.HTTP_409_CONFLICT,
                    detail="Category name already exists"
                )
            
            # Create and save
            new_category = QuizCategory(
                name=category_data.name,
                description=category_data.description,
                icon_url=str(category_data.icon_url) if category_data.icon_url else None,
                color_code=category_data.color_code or "#000000",
                created_by_id=user_id,
                is_active=True,
                is_deleted=False
            )
            db.add(new_category)
            await db.flush()
            await db.commit()
            
            # Invalidate cache
            if redis_client:
                await redis_client.delete("categories:all")
            
            logger.info(f"Category created: {new_category.id} by user {user_id}")
            
            return {
                "category_id": new_category.id,
                "name": new_category.name,
                "description": new_category.description,
                "icon_url": new_category.icon_url,
                "color_code": new_category.color_code,
                "created_at": new_category.created_at
            }
        except Exception as e:
            await db.rollback()
            logger.error(f"Category creation error: {str(e)}", exc_info=True)
            raise
```

### API Handler

```python
# app/api/v1/quiz_categories.py
from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import get_db
from app.dependencies import get_current_user
from app.models import User
from app.services.quiz_category_service import QuizCategoryService
from app.schemas.quiz_category import CategoryCreate, CategoryCreateResponse
import logging

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/api/v1/quiz-categories", tags=["Quiz Categories"])

@router.post("", response_model=CategoryCreateResponse, status_code=status.HTTP_201_CREATED)
async def create_category(
    category_data: CategoryCreate,
    current_user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db)
) -> CategoryCreateResponse:
    """Create new quiz category (admin only)"""
    try:
        # Check admin role
        if current_user.role != "admin":
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Admin access required"
            )
        
        result = await QuizCategoryService.create_category(
            db, category_data, current_user.id
        )
        
        return CategoryCreateResponse(success=True, data=result)
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Error creating category: {str(e)}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail="Failed to create category"
        )
```
