## 🗏️ Business Logic (10 Steps)

1. **Verify Admin Authentication**
   - Check JWT token has admin=true
   - Return 403 if not admin

2. **Validate Parameters**
   - page â‰¥ 1, page_size âˆˆ [1,100]
   - date_from and date_to must be valid ISO 8601
   - action_type must be in enum
   - user_id must be positive integer if provided
   - Return 400 if invalid

3. **Build Base Query**
   - SELECT activity_logs JOIN users
   - Include all activity log fields and user metadata

4. **Apply Date Range Filter**
   - If date_from: WHERE created_at >= :date_from
   - If date_to: WHERE created_at <= :date_to
   - Default: last 30 days

5. **Apply User Filter**
   - If user_id: WHERE user_id = :user_id
   - If search: WHERE username ILIKE :search OR email ILIKE :search

6. **Apply Action Type Filter**
   - If action_type != "all": WHERE action_type = :action_type
   - Default: all action types

7. **Apply Sorting**
   - If sort_by="created_at": ORDER BY created_at [ASC|DESC]
   - If sort_by="user_id": ORDER BY user_id [ASC|DESC]
   - If sort_by="action_type": ORDER BY action_type [ASC|DESC]

8. **Calculate Total Count**
   - COUNT(*) without pagination
   - For pagination info

9. **Apply Pagination**
   - offset = (page - 1) * page_size
   - LIMIT page_size OFFSET offset

10. **Return Response**
    - Include activities array with full details
    - Include pagination metadata
    - Log this admin audit access

## âŒ Error Cases (7 Scenarios)

1. **Unauthorized (401)**
   - No token or expired
   - Action: Return 401

2. **Forbidden (403)**
   - Not admin
   - Action: Return 403 with "Admin access required"

3. **Invalid Date Format (400)**
   - date_from or date_to not ISO 8601
   - Action: Return 400 with error message

4. **Invalid Date Range (400)**
   - date_from > date_to
   - Action: Return 400 with "Invalid date range"

5. **Invalid Page Size (422)**
   - page_size > 100 or < 1
   - Action: Return 422

6. **User Not Found (404)**
   - user_id filter applied but user doesn't exist
   - Action: Return 404

7. **Database Error (500)**
   - Query failure
   - Action: Return 500

### Service Layer

```python
# app/services/admin_service.py (continued)

class AdminService:
    
    @staticmethod
    async def get_activity_logs(
        page: int = 1,
        page_size: int = 50,
        date_from: Optional[datetime] = None,
        date_to: Optional[datetime] = None,
        user_id: Optional[int] = None,
        action_type: Optional[str] = None,
        search: Optional[str] = None,
        sort_by: str = "created_at",
        sort_order: str = "desc",
        db: Session = None,
        admin_id: int = None
    ) -> Dict:
        """Fetch activity logs with filtering and pagination"""
        
        # Build query
        query = db.query(ActivityLog).join(
            User, ActivityLog.user_id == User.id
        )
        
        # Apply filters
        if date_from:
            query = query.filter(ActivityLog.created_at >= date_from)
        if date_to:
            query = query.filter(ActivityLog.created_at <= date_to)
        if user_id:
            query = query.filter(ActivityLog.user_id == user_id)
        if action_type and action_type != "all":
            query = query.filter(ActivityLog.action_type == action_type)
        if search:
            search_pattern = f"%{search}%"
            query = query.filter(
                or_(
                    User.username.ilike(search_pattern),
                    User.email.ilike(search_pattern)
                )
            )
        
        # Get total count
        total_count = query.count()
        
        # Apply sorting
        if sort_by == "user_id":
            order_col = ActivityLog.user_id
        elif sort_by == "action_type":
            order_col = ActivityLog.action_type
        else:
            order_col = ActivityLog.created_at
        
        if sort_order.lower() == "asc":
            query = query.order_by(order_col.asc())
        else:
            query = query.order_by(order_col.desc())
        
        # Pagination
        offset = (page - 1) * page_size
        activities = query.offset(offset).limit(page_size).all()
        
        # Format response
        activity_list = []
        for activity in activities:
            user = db.query(User).filter(User.id == activity.user_id).first()
            activity_list.append({
                "id": activity.id,
                "user_id": activity.user_id,
                "username": user.username if user else "unknown",
                "email": user.email if user else "unknown",
                "action_type": activity.action_type,
                "action_details": activity.action_details,
                "ip_address": activity.ip_address,
                "user_agent": activity.user_agent,
                "status": activity.status,
                "created_at": activity.created_at
            })
        
        # Log admin access
        await AdminService._log_audit(
            admin_id=admin_id,
            action="VIEW_ACTIVITY_LOG",
            details={"filters": {
                "date_from": str(date_from),
                "date_to": str(date_to),
                "user_id": user_id,
                "action_type": action_type
            }},
            db=db
        )
        
        return {
            "activities": activity_list,
            "pagination": {
                "page": page,
                "page_size": page_size,
                "total_items": total_count,
                "total_pages": math.ceil(total_count / page_size),
                "has_next": page < math.ceil(total_count / page_size)
            }
        }
```

### API Handler

```python
# app/api/v1/admin.py (continued)

@router.get("/activity-log", response_model=ActivityLogResponse)
async def get_activity_log(
    page: int = Query(1, ge=1),
    page_size: int = Query(50, ge=1, le=100),
    date_from: Optional[str] = Query(None),
    date_to: Optional[str] = Query(None),
    user_id: Optional[int] = Query(None, gt=0),
    action_type: Optional[str] = Query(None),
    search: Optional[str] = Query(None, max_length=100),
    sort_by: str = Query("created_at", regex="^(created_at|user_id|action_type)$"),
    sort_order: str = Query("desc", regex="^(asc|desc)$"),
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
) -> ActivityLogResponse:
    """
    Get user activity audit log (admin only).
    
    Admin can filter by date range, user, action type, and search
    """
    try:
        if not current_user.is_admin:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Admin access required"
            )
        
        # Parse dates
        date_from_parsed = None
        date_to_parsed = None
        if date_from:
            try:
                date_from_parsed = datetime.fromisoformat(date_from)
            except:
                raise HTTPException(status_code=400, detail="Invalid date_from format")
        if date_to:
            try:
                date_to_parsed = datetime.fromisoformat(date_to)
            except:
                raise HTTPException(status_code=400, detail="Invalid date_to format")
        
        logs = await AdminService.get_activity_logs(
            page=page,
            page_size=page_size,
            date_from=date_from_parsed,
            date_to=date_to_parsed,
            user_id=user_id,
            action_type=action_type,
            search=search,
            sort_by=sort_by,
            sort_order=sort_order,
            db=db,
            admin_id=current_user.id
        )
        
        return ActivityLogResponse(success=True, data=logs)
    
    except HTTPException:
        raise
    
    except Exception as e:
        logger.error(f"Failed to fetch activity log: {str(e)}", exc_info=True)
        raise HTTPException(status_code=500, detail="Failed to retrieve activity log")
```
